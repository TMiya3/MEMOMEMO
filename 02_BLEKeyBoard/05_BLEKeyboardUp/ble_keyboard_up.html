ボタンの解放検出

26/03/21作成
ボタンを離したのを検出する
			
#include <BleKeyboard.h>

BleKeyboard bleKeyboard("ESP32 Keyboard");
const int button0Pin = 15;
const int button1Pin = 2;
bool pushed0 = false;
bool pushed1 = false;
void setup() {
  pinMode(button0Pin, INPUT_PULLUP);
  pinMode(button1Pin, INPUT_PULLUP);

  Serial.begin(115200);
  Serial.println("Starting BLE work!");
  bleKeyboard.begin();
}

void loop() {
  if (bleKeyboard.isConnected()) {
    int button0Value = digitalRead(button0Pin);
    if(button0Value == LOW){
      if(pushed0 ==false){
        bleKeyboard.print("a");
        pushed0 = true;
      }
    }
    else{
      if(pushed0){
        bleKeyboard.print("c");
      }
      pushed0 = false;
    }

    int button1Value = digitalRead(button1Pin);
    if(button1Value == LOW){
      if(pushed1 ==false){
        bleKeyboard.print("b");
        pushed1 = true;
      }
    }
    else{
      if(pushed1){
        bleKeyboard.print("d");
      }
      pushed1 = false;
    }
  }
  delay(10);
}

解説

押したフラグのリセットのタイミングが離されているとき
離されているときにフラグが立っていたら、それは離されてはじめて
なのでその時が、離された瞬間
